% MACROS

% Add Level 4 \subsubsubsection
\newcommand{\subsubsubsection}[1]{\paragraph{#1}\mbox{}\\}

% Custom centered section title
\newcommand{\centersection}[1]{
  {
  \onehalfspacing
  \phantomsection
  \begin{center}
    \normalfont\sectfont\nobreak{#1}
  \end{center}
  \vspace{-0.5ex}
  }%
}

% Three star symbol
\newcommand{\threeast}{
\begin{center}
  {$\ast\quad\ast\quad\ast$}
\end{center}
}%

\def\cpp{{C\nolinebreak[4]\hspace{-.05em}\raisebox{.4ex}{\tiny\textbf{++}}}}%

% clearpage to get to left
\newcommand*\cleartoleftpage{%
\clearpage
\ifodd\value{page}
  \vspace*{\fill}
  \thispagestyle{empty}
  \begin{center}
    \footnotesize{[Diese Seite wurde absichtlich frei gelassen.]}
  \end{center}
  \vfill
  \clearpage\newpage
\fi
}

% clearpage to get to right
\newcommand*\cleartorightpage{%
\clearpage
\ifodd\value{page}
\else
  \clearpage\null\newpage
\fi
}

% chapter heading style
\newbox\chapternumberbox
\renewcommand{\chapterlinesformat}[3]{%
  \savebox\chapternumberbox{
    \ifstr{#2}{}{}{
      \fontsize{100}{100}\normalfont\sffamily\bfseries
      \thechapter}%
    }%
    \parbox[t]{\dimexpr\linewidth-\wd\chapternumberbox-1em}{%
      \raggedleft
      \ifstr{#2}{}{}{
        \makebox[1em][l]{\normalfont\normalsize\slshape\chapapp}\\
      }%
      #3%
    }\quad
    \raisebox{-1.75\baselineskip}{\usebox\chapternumberbox}\par
}
